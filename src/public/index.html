1. Who are the first 10 authors ordered by date_of_birth?

select * from authors order by date_of_birth  asc limit 10;

2. What is the sales total for the author named “Lorelai Gilmore”?

select sum( si.item_price  * si.quantity) as total from sale_items si

inner join books b  on si.book_id  = b.id 

inner join authors a on b.author_id  = a.id 

where a.name like '%Lorelai Gilmore%'

group by a.id ;

3. What are the top 10 performing authors, ranked by sales revenue?

select count(a.id) as mostRanked, a.name from sale_items si

inner join books b  on si.book_id  = b.id 

inner join authors a on b.author_id  = a.id 

group by a.id 

order by count(a.id) desc

limit 10;
